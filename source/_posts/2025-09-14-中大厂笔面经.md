public class Main {
    public static void main(String[] args) {
        Scanner input=new Scanner(System.in);

        String strN=String.valueOf(n);
        
        //计算n的位数
        int len=strN.length();
        // System.out.println(len);

        int num=input.nextInt(); //数字集合数量
        int[] a=new int[num];
        for(int i=0;i<num;++i){
            a[i]=input.nextInt();
        }

        input.close();

        Arrays.sort(a);

        int ans=0;
        boolean flag=false;
        for(int i=0;i<len;++i){
            char c=strN.charAt(i);
            if(flag==true){
                ans=ans*10+a[num-1];
            }else{
                for(int j=num-1;j>=0;--j){
                    if(a[j]<=c-'0'){
                        if(a[j]<c-'0'){
                            flag=true;
                        }
                        ans=ans*10+a[j];
                        break;
                    }
                }
            }
        }
        System.out.println(ans);
    }
}
